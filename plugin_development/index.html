    插件开发指南¶

    本指南详细介绍如何为 Maya 悬浮工具盒开发自定义插件。

    概述¶

    工具盒的插件系统允许你： - 添加自定义工具到工具盒 - 组织工具到分类和子分类 - 为工具设置图标和描述 - 使用 Maya API 和 Python 标准库

    快速入门¶

    1. 创建插件文件¶

    在 floating_toolbox/plugins/ 目录下创建 Python 文件：

    floating_toolbox/
└── plugins/
    ├── default_tools.py      # 内置工具
    ├── my_modeling_tools.py  # 你的新插件
    └── rigging_kit.py        # 另一个插件

    2. 编写基本插件¶

    """
我的建模工具集插件
包含常用的多边形建模工具
"""
import maya.cmds as cmds


def register_plugin(manager):
    """
    注册插件工具

    这是插件的入口函数，工具盒启动时会自动调用此函数。

    Args:
        manager (ToolManager): 工具管理器实例
    """

    manager.register_tool(
        tool_id="mytools_quick_cube",    # 唯一标识符
        name="快速创建立方体",             # 显示名称
        category="建模",                  # 分类
        callback=create_quick_cube,       # 回调函数
        description="在原点创建一个 1x1x1 的立方体"
    )


def create_quick_cube():
    """创建立方体"""
    result = cmds.polyCube(
        width=1, height=1, depth=1,
        name="quickCube#"
    )
    cmds.select(result[0])
    print(f"已创建: {result[0]}")
    return result[0]

    3. 加载插件¶

    方法1：重启 Maya 并打开工具盒

    方法2：使用热重载（推荐开发时使用） - 右键点击工具盒头像 - 选择"重新加载插件"

    register_tool 参数详解¶

    manager.register_tool(
    tool_id="unique_tool_id",      # 必填：唯一标识符
    name="工具名称",                # 必填：显示名称
    category="分类名",              # 必填：工具分类
    callback=my_function,           # 必填：点击时执行的函数
    description="工具描述",         # 可选：搜索和悬停提示
    icon="icon.svg",               # 可选：图标路径
    subcategory="子分类"           # 可选：子分类名称
)

    参数说明¶

    参数 类型 必填 说明
    tool_id str ✅ 唯一标识符，建议格式：plugin_function，如 rigging_mirror_joints
    name str ✅ 工具的显示名称，支持中文
    category str ✅ 工具分类，如 "建模"、"绑定"、"动画"
    callback Callable ✅ 点击工具时执行的函数，无参数
    description str ❌ 工具描述，用于搜索匹配和鼠标悬停提示
    icon str ❌ 图标路径，支持 SVG 和 PNG
    subcategory str ❌ 子分类名称，用于分组工具

    tool_id 命名规范¶

    格式 示例 说明
    plugin_function rigging_mirror_joints 基本格式
    plugin_object_action anim_layer_create 包含对象类型
    prefix_feature_detail mytools_poly_smooth 带前缀避免冲突

    重要：tool_id 必须全局唯一，重复的 ID 会覆盖之前注册的工具。

    图标规范¶

    支持格式¶

    格式 推荐度 说明
    SVG ⭐⭐⭐ 矢量格式，支持任意缩放，推荐
    PNG ⭐⭐ 位图格式，建议 32x32 或 64x64

    路径指定¶

    # 方式1：使用内置图标（位于 resources/icons/ 目录）

# 方式3：绝对路径

# 方式4：使用 __file__ 获取相对路径

icon=os.path.join(PLUGIN_DIR, "icons", "my_tool.svg")

    设计建议¶

    • 颜色：单色深灰色 (#6272a4) 或白色线条
    • 尺寸：24x24 或 32x32 逻辑像素
    • 风格：线条图标，与 Dracula 主题协调
    • 粗细：2px 描边，保持一致性

    完整示例：绑定工具集¶

    """
Rigging Kit - 绑定工具集插件

提供一套完整的角色绑定工具，包括：
- 骨骼创建和编辑
- 控制器生成
- 蒙皮工具
- 约束工具

作者: Your Name
版本: 1.0.0
"""
import maya.cmds as cmds
import maya.mel as mel
import os

# 插件目录
PLUGIN_DIR = os.path.dirname(os.path.abspath(__file__))
ICONS_DIR = os.path.join(PLUGIN_DIR, "rigging_icons")


def register_plugin(manager):
    """注册绑定工具"""

    # ==================== 骨骼工具 ====================

    manager.register_tool(
        tool_id="rigging_create_joint",
        name="创建骨骼",
        category="绑定",
        subcategory="骨骼",
        callback=create_joint_at_selection,
        description="在选中位置或世界原点创建骨骼",
        icon=get_icon("joint.svg")
    )

    manager.register_tool(
        tool_id="rigging_orient_joints",
        name="对齐骨骼方向",
        category="绑定",
        subcategory="骨骼",
        callback=orient_joint_chain,
        description="自动对齐选中骨骼链的旋转",
        icon=get_icon("orient.svg")
    )

    manager.register_tool(
        tool_id="rigging_mirror_joints",
        name="镜像骨骼",
        category="绑定",
        subcategory="骨骼",
        callback=mirror_selected_joints,
        description="沿 YZ 平面镜像骨骼链",
        icon=get_icon("mirror.svg")
    )

    # ==================== 控制器工具 ====================

    manager.register_tool(
        tool_id="rigging_create_circle_ctrl",
        name="创建圆形控制器",
        category="绑定",
        subcategory="控制器",
        callback=lambda: create_controller("circle"),
        description="为选中骨骼创建圆形 NURBS 控制器"
    )

    manager.register_tool(
        tool_id="rigging_create_square_ctrl",
        name="创建方形控制器",
        category="绑定",
        subcategory="控制器",
        callback=lambda: create_controller("square"),
        description="为选中骨骼创建方形 NURBS 控制器"
    )

    manager.register_tool(
        tool_id="rigging_match_transforms",
        name="匹配变换",
        category="绑定",
        subcategory="控制器",
        callback=match_transforms,
        description="将第一个选中对象匹配到第二个的位置"
    )

    # ==================== 约束工具 ====================

    manager.register_tool(
        tool_id="rigging_parent_constraint",
        name="父约束",
        category="绑定",
        subcategory="约束",
        callback=create_parent_constraint,
        description="创建父约束（保持偏移）"
    )

    manager.register_tool(
        tool_id="rigging_orient_constraint",
        name="方向约束",
        category="绑定",
        subcategory="约束",
        callback=lambda: create_constraint("orient"),
        description="创建方向约束"
    )

    manager.register_tool(
        tool_id="rigging_point_constraint",
        name="点约束",
        category="绑定",
        subcategory="约束",
        callback=lambda: create_constraint("point"),
        description="创建点约束"
    )


# ==================== 辅助函数 ====================

def get_icon(name):
    """获取图标路径"""
    path = os.path.join(ICONS_DIR, name)
    return path if os.path.exists(path) else ""


def require_selection(min_count=1, node_type=None):
    """
    验证选择的装饰器

    Args:
        min_count: 最少选择数量
        node_type: 节点类型过滤
    """
    def decorator(func):
        def wrapper(*args, **kwargs):
            sel = cmds.ls(selection=True, type=node_type) if node_type else cmds.ls(selection=True)
            if len(sel) < min_count:
                type_msg = f" ({node_type})" if node_type else ""
                cmds.warning(f"请选择至少 {min_count} 个对象{type_msg}")
                return None
            return func(*args, **kwargs)
        return wrapper
    return decorator


# ==================== 骨骼工具实现 ====================

def create_joint_at_selection():
    """在选中位置创建骨骼"""
    sel = cmds.ls(selection=True)

    if sel:
        # 在选中对象位置创建
        pos = cmds.xform(sel[0], query=True, worldSpace=True, translation=True)
        cmds.select(clear=True)
        jnt = cmds.joint(position=pos, name="joint#")
    else:
        # 在世界原点创建
        cmds.select(clear=True)
        jnt = cmds.joint(position=(0, 0, 0), name="joint#")

    print(f"已创建骨骼: {jnt}")
    return jnt


@require_selection(min_count=1, node_type="joint")
def orient_joint_chain():
    """对齐骨骼方向"""
    joints = cmds.ls(selection=True, type="joint")

    for jnt in joints:
        # 检查是否有子骨骼
        children = cmds.listRelatives(jnt, children=True, type="joint")

        if children:
            # 有子骨骼：使用自动对齐
            cmds.joint(jnt, edit=True, 
                      orientJoint="xyz",
                      secondaryAxisOrient="yup",
                      zeroScaleOrient=True)
        else:
            # 末端骨骼：使用父骨骼的方向
            parent = cmds.listRelatives(jnt, parent=True, type="joint")
            if parent:
                parent_orient = cmds.getAttr(f"{parent[0]}.jointOrient")[0]
                cmds.setAttr(f"{jnt}.jointOrient", *parent_orient)

    print(f"已对齐 {len(joints)} 个骨骼")


@require_selection(min_count=1, node_type="joint")
def mirror_selected_joints():
    """镜像骨骼"""
    sel = cmds.ls(selection=True, type="joint")
    root = sel[0]

    # 使用 Maya 内置镜像
    try:
        mirrored = cmds.mirrorJoint(
            root,
            mirrorYZ=True,
            mirrorBehavior=True,
            searchReplace=("_L_", "_R_")
        )
        cmds.select(mirrored)
        print(f"已镜像骨骼: {root} -> {mirrored[0]}")
    except Exception as e:
        cmds.warning(f"镜像失败: {e}")


# ==================== 控制器工具实现 ====================

CTRL_SHAPES = {
    "circle": lambda r: cmds.circle(normal=(0, 1, 0), radius=r)[0],
    "square": lambda r: cmds.nurbsSquare(normal=(0, 1, 0), sideLength=r * 2)[0],
}


@require_selection(min_count=1)
def create_controller(shape="circle", scale=2.0):
    """为选中对象创建控制器"""
    controllers = []

    for obj in cmds.ls(selection=True):
        # 创建形状
        if shape in CTRL_SHAPES:
            ctrl = CTRL_SHAPES[shape](scale)
        else:
            ctrl = cmds.circle(normal=(0, 1, 0), radius=scale)[0]

        # 重命名
        base_name = obj.replace("_jnt", "").replace("_JNT", "")
        ctrl = cmds.rename(ctrl, f"{base_name}_ctrl")

        # 创建偏移组
        grp = cmds.group(ctrl, name=f"{ctrl}_grp")

        # 对齐到目标
        cmds.matchTransform(grp, obj)

        # 创建约束
        cmds.parentConstraint(ctrl, obj, maintainOffset=True)

        controllers.append(ctrl)

    cmds.select(controllers)
    print(f"已创建 {len(controllers)} 个控制器")
    return controllers


@require_selection(min_count=2)
def match_transforms():
    """匹配变换"""
    sel = cmds.ls(selection=True)
    target = sel[-1]  # 最后选中的是目标
    sources = sel[:-1]

    for src in sources:
        cmds.matchTransform(src, target, position=True, rotation=True)

    print(f"已将 {len(sources)} 个对象匹配到 {target}")


# ==================== 约束工具实现 ====================

@require_selection(min_count=2)
def create_parent_constraint():
    """创建父约束"""
    sel = cmds.ls(selection=True)
    driven = sel[-1]
    drivers = sel[:-1]

    constraint = cmds.parentConstraint(drivers, driven, maintainOffset=True)[0]
    print(f"已创建父约束: {constraint}")
    return constraint


@require_selection(min_count=2)
def create_constraint(constraint_type):
    """创建指定类型的约束"""
    sel = cmds.ls(selection=True)
    driven = sel[-1]
    drivers = sel[:-1]

    constraint_funcs = {
        "orient": cmds.orientConstraint,
        "point": cmds.pointConstraint,
        "scale": cmds.scaleConstraint,
        "aim": cmds.aimConstraint,
    }

    func = constraint_funcs.get(constraint_type)
    if func:
        constraint = func(drivers, driven, maintainOffset=True)[0]
        print(f"已创建 {constraint_type} 约束: {constraint}")
        return constraint
    else:
        cmds.warning(f"未知约束类型: {constraint_type}")

    1. 错误处理¶

    def my_tool():
    """正确的错误处理示例"""

    # 检查选择
    sel = cmds.ls(selection=True)
    if not sel:
        cmds.warning("请先选择对象")  # warning 不会抛出异常
        return

    try:
        # 执行可能失败的操作
        cmds.polySmooth(sel[0], divisions=2)
    except RuntimeError as e:
        # 使用 warning 而非 error
        cmds.warning(f"操作失败: {e}")
        return
    except Exception as e:
        # 记录详细错误
        import traceback
        traceback.print_exc()
        cmds.error(f"未知错误: {e}")

    2. Undo 支持¶

try:
    tool.execute()
finally:

    3. 进度条¶

    对于耗时操作，使用 Maya 进度条：

    def batch_process():
    """批量处理示例"""
    items = cmds.ls(type="mesh")
    total = len(items)

    cmds.progressWindow(
        title="批量处理",
        progress=0,
        max=total,
        isInterruptable=True
    )

    try:
        for i, item in enumerate(items):
            # 检查取消
            if cmds.progressWindow(query=True, isCancelled=True):
                break

            # 处理
            process_mesh(item)

            # 更新进度
            cmds.progressWindow(
                edit=True,
                progress=i + 1,
                status=f"处理中: {item}"
            )
    finally:
        cmds.progressWindow(endProgress=True)

    4. 线程安全¶

    Maya 命令只能在主线程执行。如需异步操作：

    import maya.utils

def my_async_tool():
    """在后台线程处理，结果在主线程应用"""
    import threading

    def background_work():
        # 计算密集型工作
        result = heavy_computation()

        # 回到主线程执行 Maya 命令
        maya.utils.executeDeferred(lambda: apply_result(result))

    thread = threading.Thread(target=background_work)
    thread.start()

    常见问题¶

    Q1: 插件没有加载？¶

    检查步骤： 1. 确认文件在 plugins/ 目录下 2. 确认有 register_plugin(manager) 函数 3. 检查 Maya Script Editor 的错误信息 4. 确认 Python 语法正确（无 IndentationError 等）

    Q2: 工具没有显示？¶

    可能原因： - tool_id 重复，被其他工具覆盖 - 分类名称有隐藏空格 - callback 函数有语法错误

    Q3: 图标不显示？¶

    检查： - 路径是否正确（使用 os.path.exists() 验证） - 文件格式是否为 SVG 或 PNG - 相对路径是否相对于正确的目录

    Q4: 工具执行报错？¶

    调试方法： 

    import traceback

def my_tool():
    try:
        # 工具逻辑
        pass
    except Exception as e:
        traceback.print_exc()  # 打印完整堆栈
        cmds.warning(str(e))

import maya.cmds as cmds
import os

PLUGIN_DIR = os.path.dirname(os.path.abspath(__file__))


def register_plugin(manager):
    """注册工具"""

    manager.register_tool(
        tool_id="myplugin_tool_1",
        name="工具1",
        category="我的分类",
        subcategory="子分类",
        callback=tool_1_function,
        description="工具1的描述",
        icon=os.path.join(PLUGIN_DIR, "icons", "tool1.svg")
    )


def tool_1_function():
    """工具1的实现"""
    sel = cmds.ls(selection=True)
    if not sel:
        cmds.warning("请选择对象")
        return

    # 实现工具逻辑
    result = do_something(sel)
    print(f"完成: {result}")
    return result


def do_something(objects):
    """辅助函数"""
    return len(objects)
